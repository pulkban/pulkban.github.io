Show input value as user types [2-way Input Binding] 


import React, { useState, useEffect } from 'react';

function App() {
  const [users, setUsers] = useState();

  useEffect(() => {
    fetch('https://jsonplaceholder.typicode.com/users')
      .then(res => res.json)
      .then(data => {
        setUsers(data);
      });
  });

  return (
    <div>
      <h2>User List</h2>
      {users.map(user => (
        <p>{user.name}</p>
      ))}
      <button onClick={handleClick()}>Click</button>
    </div>
  );
}

export default App;
