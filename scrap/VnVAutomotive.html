--Write a Python script to parse a CAN log and count the number of messages for each CAN ID.
Input:
100 01 02 03
200 11 22 33
100 AA BB CC
300 55 66
100 FF EE
Expected Output:
100 : 3
200 : 1
300 : 1
Code:
counts = {}
with open("canlog.txt") as f:
    for line in f:
        can_id = line.split()[0]
        counts[can_id] = counts.get(can_id, 0) + 1
for k, v in counts.items():
    print(k, v)

--Read a log file and count the number of ERROR lines.
count = 0
with open("log.txt") as f:
    for line in f:
        if "ERROR" in line:
            count += 1
print(count)
