import pytest
from selenium import webdriver
from selenium.webdriver.common.by import By

@pytest.fixture
def driver():
    driver = webdriver.Chrome()
    driver.get("https://example.com/login")
    return driver
    driver.quit()

def test_login(driver):

    username = driver.find_element(By.ID,"username")
    password = driver.find_element(By.ID,"password")
    loginBtn = driver.find_element(By.ID,"loginBtn")

    username.send_keys("testuser")
    password.send_keys("testpwd")
    loginBtn.click

    welcome_text = driver.find_element(By.ID,"welcomeMsg")

    assert welcome_text = "Welcome User"

import pytest
from selenium import webdriver
from selenium.webdriver.common.by import By

@pytest.fixture
def driver():
    driver = webdriver.Chrome()
    driver.get("https://shop.example.com")
    return driver
    driver.quit()

def test_order_flow(driver):

    search = driver.find_element(By.ID,"searchBox")
    search.send_keys("Laptop")
    driver.find_element(By.ID,"searchBtn").click

    product = driver.find_element(By.XPATH,"//div[@class='product'][1]")
    product.click()

    driver.find_element(By.ID,"addCart").click()

    driver.find_element(By.ID,"cart").click()
    driver.find_element(By.ID,"checkoutBtn").click()

    card = driver.find_element(By.ID,"cardNumber")
    card.send_keys(1234567890123456)

    driver.find_element(By.ID,"payBtn").click()

    order = driver.find_element(By.ID,"orderNo")

    assert order = "Order Confirmed"

    logout = driver.find_element(By.ID,"logout")
    logout.click()
